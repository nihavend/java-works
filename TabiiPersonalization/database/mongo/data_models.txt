func FillImage(image *mongo.Fields, images map[string]dynamo.Content, imageType string, key string) {
	if image != nil {
		images[imageType] = dynamo.Content{
			ContentType: image.Type.(string),
			Name:        image.FileName,
			Title:       image.Title,
			ImageType:   imageType,
			Key:         key,
		}
	}
}

CREATE TABLE public.images (
	id serial4 NOT NULL,
	image_type text NULL,
	filename text NULL,
	title text NULL,
	CONSTRAINT images_filename_key UNIQUE (filename),
	CONSTRAINT images_pkey PRIMARY KEY (id)
);    


CREATE TABLE public.content_images (
	content_id int8 NOT NULL,
	image_id int4 NOT NULL,
	CONSTRAINT content_images_pkey PRIMARY KEY (content_id, image_id)
);
